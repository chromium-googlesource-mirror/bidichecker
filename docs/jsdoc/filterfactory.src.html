 85 /**
 86  * Constructs a suppression filter from a bare object with suitable contents.
 87  * For use in deserializing filters from JSON. Throws a string exception if an
 88  * invalid filter is encountered.
 89  * @param {!Object} bareFilter The object representing the filter. Must contain
 90  *     an 'opcode' field with a string indicating the filter type.
 91  *     Other supported field names and types are dependent on the filter type.
 92  * @return {!bidichecker.FilterFactory.ComposableFilter_} The new filter.
 93  */
 94 bidichecker.FilterFactory.constructFilter = function(bareFilter) {
 95   var opcode = bareFilter['opcode'];
 96   switch (opcode) {
 97     case 'AND':
 98       return new bidichecker.FilterFactory.AndFilter_(bareFilter);
 99 
100     case 'AT_TEXT':
101       return new bidichecker.FilterFactory.AtTextFilter_(bareFilter);
102 
103     case 'AT_TEXT_REGEXP':
104       return new bidichecker.FilterFactory.AtTextRegexpFilter_(bareFilter);
105 
106     case 'FOLLOWED_BY_TEXT':
107       return new bidichecker.FilterFactory.FollowedByTextFilter_(bareFilter);
108 
109     case 'FOLLOWED_BY_TEXT_REGEXP':
110       return new bidichecker.FilterFactory.FollowedByTextRegexpFilter_(
111           bareFilter);
112 
113     case 'LOCATION_CLASS':
114       return new bidichecker.FilterFactory.LocationClassFilter_(bareFilter);
115 
116     case 'LOCATION_CLASS_REGEXP':
117       return new bidichecker.FilterFactory.LocationClassRegexpFilter_(
118           bareFilter);
119 
120     case 'LOCATION_ID':
121       return new bidichecker.FilterFactory.LocationIdFilter_(bareFilter);
122 
123     case 'LOCATION_ID_REGEXP':
124       return new bidichecker.FilterFactory.LocationIdRegexpFilter_(
125           bareFilter);
126 
127     case 'NOT':
128       return new bidichecker.FilterFactory.NotFilter_(bareFilter);
129 
130     case 'OR':
131       return new bidichecker.FilterFactory.OrFilter_(bareFilter);
132 
133     case 'PRECEDED_BY_TEXT':
134       return new bidichecker.FilterFactory.PrecededByTextFilter_(bareFilter);
135 
136     case 'PRECEDED_BY_TEXT_REGEXP':
137       return new bidichecker.FilterFactory.PrecededByTextRegexpFilter_(
138           bareFilter);
139 
140     case 'SEVERITY':
141       return new bidichecker.FilterFactory.SeverityFilter_(bareFilter);
142 
143     case 'TYPE':
144       return new bidichecker.FilterFactory.TypeFilter_(bareFilter);
145 
146     default:
147       throw 'Unknown filter opcode: \'' + opcode + '\'';
148   }
149 };

152 /**
153  * Deserializes a list of suppression filters from a JSON-format string.
154  * Throws a string exception if an unknown filter is encountered.
155  * @param {string=} opt_filtersJson The input string, if provided.
156  * @return {Array.<!bidichecker.Filter>} The error suppression filters.
157  */
158 bidichecker.FilterFactory.readFiltersFromJson = function(opt_filtersJson) {
159   if (!opt_filtersJson) {
160     return null;
161   }
162 
163   var bareFilters =
164       (/** @type {Array.<!Object>} */ goog.json.parse(opt_filtersJson));
165   return goog.array.map(bareFilters, function(bareFilter) {
166     return bidichecker.FilterFactory.constructFilter(bareFilter);
167   });
168 };

396 /**
397  * Checks that a required field exists in an object representing a filter, or
398  * else throws a string exception.
399  * @param {!Object} bareObject The object representing the filter.
400  * @param {string} field The name of the field.
401  * @private
402  */
403 bidichecker.FilterFactory.checkParamExists_ = function(bareObject, field) {
404   if (!(field in bareObject)) {
405     throw 'No \'' + field + '\' parameter found for \'' +
406         bareObject['opcode'] + '\' filter';
407   }
408 };

411 /**
412  * Checks that a required field of a given type exists in an object representing
413  * a filter, or else throws a string exception.
414  * @param {!Object} bareObject The object representing the filter.
415  * @param {string} field The name of the field.
416  * @param {string} type The required type of the field.
417  * @private
418  */
419 bidichecker.FilterFactory.checkParam_ = function(bareObject, field, type) {
420   bidichecker.FilterFactory.checkParamExists_(bareObject, field);
421   if (typeof bareObject[field] != type) {
422     throw 'Wrong type for \'' + field + '\' parameter of \'' +
423         bareObject['opcode'] + '\' filter; expected ' + type;
424   }
425 };

454 /**
455  * Reads a regular expression field from an object representing a filter.
456  * The field contents must be a string or a RegExp object.
457  * @param {!Object} bareObject The object representing the filter.
458  * @param {string} fieldName The name of the field.
459  * @return {!RegExp} A regular expression anchored at both ends of the string.
460  * @private
461  */
462 bidichecker.FilterFactory.getRegexpParam_ = function(bareObject, fieldName) {
463   bidichecker.FilterFactory.checkParamExists_(bareObject, fieldName);
464   var field = bareObject[fieldName];
465   var pattern;
466   if (typeof field == 'string') {
467     pattern = (/** @type {string} */ field);
468   } else if (field instanceof RegExp) {
469     pattern = field.source;
470   } else {
471     throw 'Wrong type for \'' + fieldName + '\' parameter of \'' +
472         bareObject['opcode'] + '\' filter; expected string or RegExp';
473   }
474   return new RegExp('^(' + pattern + ')$');
475 };

478 /**
479  * Reads a field representing a subfilter from an object representing a filter.
480  * The field contents can either be a bare object representing a serialized
481  * filter, or an object supporting an isSuppressed() method.
482  * @param {!Object} bareObject The object representing the filter.
483  * @param {string} field The name of the field.
484  * @return {!bidichecker.FilterFactory.ComposableFilter_} A filter constructed
485  *     from the field's contents.
486  * @private
487  */
488 bidichecker.FilterFactory.getFilterParam_ = function(
489     bareObject, field) {
490   bidichecker.FilterFactory.checkParam_(bareObject, field, 'object');
491 
492   // Recursively build any subfilters referred to by this one.
493   var subfilterObject = bareObject[field];
494   if (subfilterObject instanceof bidichecker.FilterFactory.ComposableFilter_) {
495     // Found a constructed filter object.
496     return subfilterObject;
497   } else if (typeof subfilterObject['opcode'] == 'string') {
498     // Found a raw filter object.
499     return bidichecker.FilterFactory.constructFilter(subfilterObject);
500   } else {
501     throw 'Can\'t make a filter out of the \'' + field + '\' parameter of \'' +
502         bareObject['opcode'] + '\' filter';
503   }
504 };

507 /**
508  * A filter which suppresses errors by and-ing component filters.
509  * @param {!Object} bareObject An object containing 'filter1' and 'filter2'
510  *     fields, each containing a bare object representing a component filter.
511  * @constructor
512  * @extends {bidichecker.FilterFactory.ComposableFilter_}

531     bidichecker.FilterFactory.ComposableFilter_);
