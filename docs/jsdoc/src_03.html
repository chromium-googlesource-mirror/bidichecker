 79 /**
 80  * Constructs a suppression filter from a bare object with suitable contents.
 81  * For use in deserializing filters from JSON. Throws a string exception if an
 82  * invalid filter is encountered.
 83  * @param {!Object} bareFilter The object representing the filter. Must contain
 84  *     an 'opcode' field with a string indicating the filter type.
 85  *     Other supported field names and types are dependent on the filter type.
 86  * @return {!bidichecker.FilterFactory.ComposableFilter} The new filter.
 87  * @private
 88  */
 89 bidichecker.FilterFactory.constructFilter_ = function(bareFilter) {
 90   var opcode = bareFilter['opcode'];
 91   switch (opcode) {
 92     case 'AND':
 93       return new bidichecker.FilterFactory.AndFilter_(bareFilter);
 94 
 95     case 'AT_TEXT':
 96       return new bidichecker.FilterFactory.AtTextFilter_(bareFilter);
 97 
 98     case 'AT_TEXT_REGEXP':
 99       return new bidichecker.FilterFactory.AtTextRegexpFilter_(bareFilter);
100 
101     case 'FOLLOWED_BY_TEXT':
102       return new bidichecker.FilterFactory.FollowedByTextFilter_(bareFilter);
103 
104     case 'FOLLOWED_BY_TEXT_REGEXP':
105       return new bidichecker.FilterFactory.FollowedByTextRegexpFilter_(
106           bareFilter);
107 
108     case 'LOCATION_CLASS':
109       return new bidichecker.FilterFactory.LocationClassFilter_(bareFilter);
110 
111     case 'LOCATION_CLASS_REGEXP':
112       return new bidichecker.FilterFactory.LocationClassRegexpFilter_(
113           bareFilter);
114 
115     case 'LOCATION_ID':
116       return new bidichecker.FilterFactory.LocationIdFilter_(bareFilter);
117 
118     case 'LOCATION_ID_REGEXP':
119       return new bidichecker.FilterFactory.LocationIdRegexpFilter_(
120           bareFilter);
121 
122     case 'NOT':
123       return new bidichecker.FilterFactory.NotFilter_(bareFilter);
124 
125     case 'OR':
126       return new bidichecker.FilterFactory.OrFilter_(bareFilter);
127 
128     case 'PRECEDED_BY_TEXT':
129       return new bidichecker.FilterFactory.PrecededByTextFilter_(bareFilter);
130 
131     case 'PRECEDED_BY_TEXT_REGEXP':
132       return new bidichecker.FilterFactory.PrecededByTextRegexpFilter_(
133           bareFilter);
134 
135     case 'SEVERITY':
136       return new bidichecker.FilterFactory.SeverityFilter_(bareFilter);
137 
138     case 'TYPE':
139       return new bidichecker.FilterFactory.TypeFilter_(bareFilter);
140 
141     default:
142       throw 'Unknown filter opcode: \'' + opcode + '\'';
143   }
144 };

147 /**
148  * Deserializes a list of suppression filters from a JSON-format string.
149  * Throws a string exception if an unknown filter is encountered.
150  * @param {string=} opt_filtersJson The input string, if provided.
151  * @return {Array.<!bidichecker.Filter>} The error suppression filters.
152  */
153 bidichecker.FilterFactory.readFiltersFromJson = function(opt_filtersJson) {
154   if (!opt_filtersJson) {
155     return null;
156   }
157 
158   var bareFilters =
159       (/** @type {Array.<!Object>} */ goog.json.parse(opt_filtersJson));
160   return goog.array.map(bareFilters, function(bareFilter) {
161     return bidichecker.FilterFactory.constructFilter_(bareFilter);
162   });
163 };

391 /**
392  * Checks that a required field exists in an object representing a filter, or
393  * else throws a string exception.
394  * @param {!Object} bareObject The object representing the filter.
395  * @param {string} field The name of the field.
396  * @private
397  */
398 bidichecker.FilterFactory.checkParamExists_ = function(bareObject, field) {
399   if (!(field in bareObject)) {
400     throw 'No \'' + field + '\' parameter found for \'' +
401         bareObject['opcode'] + '\' filter';
402   }
403 };

406 /**
407  * Checks that a required field of a given type exists in an object representing
408  * a filter, or else throws a string exception.
409  * @param {!Object} bareObject The object representing the filter.
410  * @param {string} field The name of the field.
411  * @param {string} type The required type of the field.
412  * @private
413  */
414 bidichecker.FilterFactory.checkParam_ = function(bareObject, field, type) {
415   bidichecker.FilterFactory.checkParamExists_(bareObject, field);
416   if (typeof bareObject[field] != type) {
417     throw 'Wrong type for \'' + field + '\' parameter of \'' +
418         bareObject['opcode'] + '\' filter; expected ' + type;
419   }
420 };

449 /**
450  * Reads a regular expression field from an object representing a filter.
451  * The field contents must be a string or a RegExp object.
452  * @param {!Object} bareObject The object representing the filter.
453  * @param {string} fieldName The name of the field.
454  * @return {!RegExp} A regular expression anchored at both ends of the string.
455  * @private
456  */
457 bidichecker.FilterFactory.getRegexpParam_ = function(bareObject, fieldName) {
458   bidichecker.FilterFactory.checkParamExists_(bareObject, fieldName);
459   var field = bareObject[fieldName];
460   var pattern;
461   if (typeof field == 'string') {
462     pattern = (/** @type {string} */ field);
463   } else if (field instanceof RegExp) {
464     pattern = field.source;
465   } else {
466     throw 'Wrong type for \'' + fieldName + '\' parameter of \'' +
467         bareObject['opcode'] + '\' filter; expected string or RegExp';
468   }
469   return new RegExp('^(' + pattern + ')$');
470 };

473 /**
474  * Reads a field representing a subfilter from an object representing a filter.
475  * The field contents can either be a bare object representing a serialized
476  * filter, or an object supporting an isSuppressed() method.
477  * @param {!Object} bareObject The object representing the filter.
478  * @param {string} field The name of the field.
479  * @return {!bidichecker.FilterFactory.ComposableFilter} A filter constructed
480  *     from the field's contents.
481  * @private
482  */
483 bidichecker.FilterFactory.getFilterParam_ = function(
484     bareObject, field) {
485   bidichecker.FilterFactory.checkParam_(bareObject, field, 'object');
486 
487   // Recursively build any subfilters referred to by this one.
488   var subfilterObject = bareObject[field];
489   if (subfilterObject instanceof bidichecker.FilterFactory.ComposableFilter) {
490     // Found a constructed filter object.
491     return subfilterObject;
492   } else if (typeof subfilterObject['opcode'] == 'string') {
493     // Found a raw filter object.
494     return bidichecker.FilterFactory.constructFilter_(subfilterObject);
495   } else {
496     throw 'Can\'t make a filter out of the \'' + field + '\' parameter of \'' +
497         bareObject['opcode'] + '\' filter';
498   }
499 };

682     for (var current = locationElements[i]; current;
683          current = current.parentNode) {
684       if (current.className) {
685         // Match each whitespace-separated class name separately.
686         var classNames = current.className.split(/\s/);

693     }
694   }
